CREATE TABLE STUDENTS(
    STUID INT NOT NULL,
    GENDER CHAR(1) NOT NULL,
    ETHNICITY CHAR(50) NOT NULL,
    CITY CHAR(50),
    DISTRICT CHAR(50)NOT NULL,
    FEEDER CHAR(50)NOT NULL,
    PROGRAMSTART DATE NOT NULL,
    PROGRAMSTATUS CHAR(25) NOT NULL,
    GRADDATE DATE,
    PROGRAMEND DATE,
    PRIMARY KEY (STUID)
);

CREATE TABLE COURSES(
    COURSECODEID CHAR(25) NOT NULL,
    COURSETITLE CHAR(100) NOT NULL,
    PRIMARY KEY (COURSECODEID)
);

CREATE TABLE GRADES(
    COURSEGRADEID INT NOT NULL,
    STUID INT NOT NULL,
    COURSECODEID CHAR(25) NOT NULL,
    COURSEGRADE CHAR(25),
    PRIMARY KEY (COURSEGRADEID),
    FOREIGN KEY (STUID) 
    REFERENCES STUDENTS(STUID),
    FOREIGN KEY (COURSECODEID)
    REFERENCES COURSES(COURSECODEID)
);
